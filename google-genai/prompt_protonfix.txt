from protonfixes import util
from protonfixes import log
import os
import configgen.utils.popup as popup #for pixL popup message

def main():
    """
    Main function for applying TeknoParrot/SR3 specific fixes.
    """
    # Install dependencies
    popup.Message("[GameFix] vcrun2010 installation/check...", 3*60) # usually a minimum of 3 minutes of display for this message
    util.protontricks('vcrun2010')
    popup.CleanLast()
    popup.Message("[GameFix] vcrunXXXX installation/check...", 3*60) # usually a minimum of 3 minutes of display for this message
    util.protontricks('vcrunXXXX')
    popup.CleanLast()
    
    popup.Message("[GameFix] Direct 3D compilers installation/check...", 3*60) # usually a minimum  3 minutes of display for this message
    util.protontricks('d3dcompiler_42')
    util.winedll_override('d3dcompiler_42', 'n,b')
    util.protontricks('d3dcompiler_XX')
    util.winedll_override('d3dcompiler_XX', 'n,b' or 'b,n' or 'b' or 'n' or 'd')
    popup.CleanLast()

    popup.Message("[GameFix] set additional Dll overrides...", 1*60) # usually 1 minute of display for all averrides for this message
    util.winedll_override('xxxxx', 'n,b')
    popup.CleanLast()

    popup.Message("[GameFix] set specific environement variables...", 1*60) # usually 1 minute of display for all averrides for this message
